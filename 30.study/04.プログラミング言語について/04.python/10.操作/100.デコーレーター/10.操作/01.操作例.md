#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import functools

"""
文字を大文字にしたり
関数の名前と引数を出よくさせたりする
デコレーターの作成例
"""

############ define decoraters ###############
"""
decorator は
デコレートした関数に影響を与えるために、
引数として受け取った関数を
closure で wrap する必要がある
"""
# わしは、とりついた関数を


def uppercase(func):

    @functools.wraps(func)  # こいつは、元の関数のメタデータを引き継ぐのに使える __doc__ とか
    # ネストした関数で
    def wrapper():
        # こうやって
        original_result = func()
        # こうやって加工して
        modified_result = original_result.upper()
        # 加工した値を返す
        return modified_result
    # そんな機能のネスト(wrap)した関数(closure)を返すんやで！
    return wrapper


def strong(func):
    def wrapper():
        return '<strong>' + func() + '</strong>'
    return wrapper


def emphasis(func):
    def wrapper():
        return '<em>' + func() + '</em>'
    return wrapper
############ define decoraters ###############

############ use decoraters ###############
@uppercase
@strong
@emphasis
def hello():
    return "hello"
############ use decoraters ###############






############ define decoraters ###############
def proxy(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper


def trace_name_and_args(func):
    def wrapper(*args, **kwargs):
        print("TRACE: calling {}() with {}, {}".format(
            func.__name__, args, kwargs))

        original_result = func(*args, **kwargs)

        print("TRACE: {}() returned {}".format(func.__name__, original_result))

        return original_result
    return wrapper
############ define decoraters ###############


@trace_name_and_args
@proxy
def say(name, line):
    return "{}: {}".format(name, line)


if __name__ == "__main__":
    # proxy の用途はわからん
    print(say("Shibata", "I was not wrong!"))
    print(hello())
